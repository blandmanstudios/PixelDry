

# collect pngs to videos
cd raw_frames/second_numbers_at_24fps

# write 24 images per second to 30 fps video
ffmpeg -framerate 24 -pattern_type glob -i '*.png' -c:v libx264 -r 30 -pix_fmt yuv420p output.mp4
# write 1 image per second to 30fps video
ffmpeg -framerate 1 -pattern_type glob -i '*.png' -c:v libx264 -r 30 -pix_fmt yuv420p output.mp4
